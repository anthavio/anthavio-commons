80  		public ClassLoadProcessor(String packageName, ClassLoader classLoader) {
81  			if (classLoader == null) {
82  				throw new IllegalArgumentException("ClassLoader is null");
83  			}
84  			URL baseUrl = classLoader.getResource(packageName.replace('.', '/'));
85  			if (baseUrl == null) {
86  				throw new IllegalArgumentException("Package not found " + packageName);
87  			}
88  			if (isJarURL(baseUrl)) {
89  				idxPrefix = baseUrl.getFile().indexOf("!/") + 2;
90  			} else {
91  				idxPrefix = baseUrl.getFile().length() - packageName.length();
92  			}
93  			this.classLoader = classLoader;
94  		}
95  
96  		public Class<?> process(URL url) {
97  			String className = url.getFile().substring(idxPrefix, url.getFile().length() - 6).replace('/', '.');
98  			try {
99  				return classLoader.loadClass(className);
100 			} catch (ClassNotFoundException cnfx) {
101 				throw new IllegalArgumentException(cnfx);
102 			}
103 		}
104 	}

196 	private static <T> void recurse(File file, List<T> ret, ResourceProcessor<T> processor) {
197 		if (file.isDirectory()) {
198 			File[] list = file.listFiles();
199 			for (File filex : list) {
200 				recurse(filex, ret, processor);
201 			}
202 		} else {
203 			try {
204 				T result = processor.process(file.toURI().toURL());
205 				if (result != null) {
206 					ret.add(result);
207 				}
208 			} catch (MalformedURLException mux) {
209 				throw new IllegalArgumentException("File can't be converted to URL");
210 			}
211 		}

214 	public static List<URL> list(URL baseUrl, NameFilter filter) {
215 		if (baseUrl == null) {
216 			throw new IllegalArgumentException("URL is null");
217 		}
218 		if (filter == null) {
219 			throw new IllegalArgumentException("NameFilter is null");
220 		}
221 		List<URL> ret = new ArrayList<URL>();
222 		try {
223 			if (isJarURL(baseUrl)) {
224 				//URL - jar:file:/C:/tmp/spring-core-3.0.5.RELEASE.jar!/org/springframework/util/comparator
225 				String jarContentPrefix = baseUrl.getFile().substring(baseUrl.getFile().indexOf("!/") + 2);
226 				String strBaseUrl = baseUrl.toString();
227 				if (strBaseUrl.endsWith("/") == false) {
228 					strBaseUrl += "/";
229 				}
230 				JarInputStream jis = null;
231 				try {
232 					URL jarFileUrl = extractJarFileURL(baseUrl);
233 					jis = new JarInputStream(jarFileUrl.openStream());
234 					JarEntry jarEntry;
235 					while ((jarEntry = jis.getNextJarEntry()) != null) {
236 						String jarEntryName = jarEntry.getName();
237 						//We are traversing whole jar file content, so have to skip everything that it not inside jarContentPrefix
238 						if (jarEntryName.startsWith(jarContentPrefix) && jarEntry.isDirectory() == false) {
239 							int idxLastSlash = jarEntryName.lastIndexOf('/');
240 							String jarEntryPath = jarEntryName.substring(0, idxLastSlash);
241 							String simpleName = jarEntryName.substring(idxLastSlash + 1); //without slash
242 							if (filter.matches(jarEntryPath, simpleName)) {
243 								ret.add(new URL(strBaseUrl + simpleName));
244 							}
245 						}
246 					}
247 
248 				} finally {
249 					if (jis != null) {
250 						try {
251 							jis.close();
252 						} catch (IOException iox) {
253 							//ignore this one to not forget potential existing exception
254 						}
255 					}
256 				}
257 			} else {
258 				File fileDir = getFile(baseUrl);
259 				recurse(fileDir, ret, filter);
260 			}
261 		} catch (IOException iox) {
262 			throw new IllegalArgumentException("Failed to list " + baseUrl, iox);
263 		}
264 		return ret;
265 	}

267 	private static void recurse(File file, List<URL> ret, NameFilter filter) throws MalformedURLException {
268 		if (file.isDirectory()) {
269 			File[] list = file.listFiles();
270 			for (File filex : list) {
271 				recurse(filex, ret, filter);
272 			}
273 		} else {
274 			if (filter.matches(file.getParent(), file.getName())) {
275 				ret.add(file.toURI().toURL());
276 			}
277 		}
278 	}

309 	/**
310 	 * Determine whether the given URL points to a resource in a jar file,
311 	 * that is, has protocol "jar", "zip", "wsjar" or "code-source".
312 	 * <p>"zip" and "wsjar" are used by BEA WebLogic Server and IBM WebSphere, respectively,
313 	 * but can be treated like jar files. The same applies to "code-source" URLs on Oracle
314 	 * OC4J, provided that the path contains a jar separator.
315 	 * @param url the URL to check
316 	 * @return whether the URL has been identified as a JAR URL
317 	 */
318 	public static boolean isJarURL(URL url) {
319 		String protocol = url.getProtocol();
320 		return (URL_PROTOCOL_JAR.equals(protocol) || URL_PROTOCOL_ZIP.equals(protocol)
321 				|| URL_PROTOCOL_WSJAR.equals(protocol) || (URL_PROTOCOL_CODE_SOURCE.equals(protocol) && url.getPath().contains(
322 						JAR_URL_SEPARATOR)));
323 	}

326 	 * Extract the URL for the actual jar file from the given URL
327 	 * (which may point to a resource in a jar file or to a jar file itself).
328 	 * @param jarUrl the original URL
329 	 * @return the URL for the actual jar file
330 	 * @throws MalformedURLException if no valid jar file URL could be extracted
331 	 */
332 	public static URL extractJarFileURL(URL jarUrl) throws MalformedURLException {
333 		String urlFile = jarUrl.getFile();
334 		int separatorIndex = urlFile.indexOf(JAR_URL_SEPARATOR);
335 		if (separatorIndex != -1) {
336 			String jarFile = urlFile.substring(0, separatorIndex);
337 			try {
338 				return new URL(jarFile);
339 			} catch (MalformedURLException ex) {
340 				// Probably no protocol in original jar URL, like "jar:C:/mypath/myjar.jar".
341 				// This usually indicates that the jar file resides in the file system.
342 				if (!jarFile.startsWith("/")) {
343 					jarFile = "/" + jarFile;
344 				}
345 				return new URL(FILE_URL_PREFIX + jarFile);
346 			}
347 		} else {
348 			return jarUrl;
349 		}
350 	}

364 	/**
365 	 * Resolve the given resource URL to a <code>java.io.File</code>,
366 	 * i.e. to a file in the file system.
367 	 * @param resourceUrl the resource URL to resolve
368 	 * @param description a description of the original resource that
369 	 * the URL was created for (for example, a class path location)
370 	 * @return a corresponding File object
371 	 * @throws FileNotFoundException if the URL cannot be resolved to
372 	 * a file in the file system
373 	 */
374 	public static File getFile(URL resourceUrl, String description) throws FileNotFoundException {
375 		if (resourceUrl == null) {
376 			throw new IllegalArgumentException("Resource URL must not be null");
377 		}
378 		if (!URL_PROTOCOL_FILE.equals(resourceUrl.getProtocol())) {
379 			throw new FileNotFoundException(description + " cannot be resolved to absolute file path "
380 					+ "because it does not reside in the file system: " + resourceUrl);
381 		}
382 		try {
383 			return new File(toURI(resourceUrl).getSchemeSpecificPart());
384 		} catch (URISyntaxException ex) {
385 			// Fallback for URLs that are not valid URIs (should hardly ever happen).
386 			return new File(resourceUrl.getFile());
387 		}
388 	}

416 	 * Replace all occurences of a substring within a string with
417 	 * another string.
418 	 * @param inString String to examine
419 	 * @param oldPattern String to replace
420 	 * @param newPattern String to insert
421 	 * @return a String with the replacements
422 	 */
423 	public static String replace(String inString, String oldPattern, String newPattern) {
424 		if (!hasLength(inString) || !hasLength(oldPattern) || newPattern == null) {
425 			return inString;
426 		}
427 		StringBuilder sb = new StringBuilder();
428 		int pos = 0; // our position in the old string
429 		int index = inString.indexOf(oldPattern);
430 		// the index of an occurrence we've found, or -1
431 		int patLen = oldPattern.length();
432 		while (index >= 0) {
433 			sb.append(inString.substring(pos, index));
434 			sb.append(newPattern);
435 			pos = index + patLen;
436 			index = inString.indexOf(oldPattern, pos);
437 		}
438 		sb.append(inString.substring(pos));
439 		// remember to append any characters to the right of a match
440 		return sb.toString();
441 	}
