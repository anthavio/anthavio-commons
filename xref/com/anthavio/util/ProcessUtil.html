15  public class ProcessUtil {

17  	/** 
18  	 * Sun property pointing the main class and its arguments. 
19  	 * Might not be defined on non Hotspot VM implementations.
20  	 */
21  	public static final String SUN_JAVA_COMMAND = "sun.java.command";
22  
23  	/**
24  	 * XXX A co treba {@link java.lang.ProcessBuilder} 
25  	 * Restart the current Java application
26  	 * @param runBeforeRestart some custom code to be run before restarting
27  	 */
28  	public static void restart(Runnable runBeforeRestart) {
29  		String javaPath = System.getProperty("java.home") + "/bin/java";
30  		List<String> vmArgsList = ManagementFactory.getRuntimeMXBean().getInputArguments();
31  		StringBuffer vmArgs = new StringBuffer();
32  		for (String vmArg : vmArgsList) {
33  			// if it's the agent argument : we ignore it otherwise the
34  			// address of the old application and the new one will be in conflict
35  			if (!vmArg.contains("-agentlib")) {
36  				vmArgs.append(vmArg);
37  				vmArgs.append(" ");
38  			}
39  		}
40  		// init the command to execute, add the vm args
41  		final StringBuffer cmd = new StringBuffer("\"" + javaPath + "\" " + vmArgs);
42  
43  		// program main and program arguments
44  		String[] mainCommand = System.getProperty(SUN_JAVA_COMMAND).split(" ");
45  		if (mainCommand[0].endsWith(".jar")) {
46  			cmd.append("-jar " + new File(mainCommand[0]).getPath());
47  		} else {
48  			cmd.append("-cp \"" + System.getProperty("java.class.path") + "\" " + mainCommand[0]);
49  		}
50  		// finally add program arguments
51  		for (int i = 1; i < mainCommand.length; i++) {
52  			cmd.append(" ");
53  			cmd.append(mainCommand[i]);
54  		}
55  		// execute the command in a shutdown hook, to be sure that all the
56  		// resources have been disposed before restarting the application
57  		Runtime.getRuntime().addShutdownHook(new Thread() {
58  			@Override
59  			public void run() {
60  				try {
61  					Runtime.getRuntime().exec(cmd.toString());
62  				} catch (IOException e) {
63  					e.printStackTrace();
64  				}
65  			}
66  		});
67  
68  		if (runBeforeRestart != null) {
69  			runBeforeRestart.run();
70  		}
71  		System.exit(0);
72  	}
73  }
