68  		DateFormat formatter = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss");
69  		DateFormat tzFormatter = new SimpleDateFormat("Z");
70  		String timezone = tzFormatter.format(dt);

75  	 * xsd:dateTime           2009-12-06T15:59:34+01:00
76  	 * yyyy-MM-dd'T'HH:mm:ssZ 2009-12-06T15:59:34+0100
77  	 */
78  	public static Date parseXsdDateTime(String xsdDateTime) {
79  		DateFormat sdf = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss");
80  
81  		Matcher mat = TZ_REGEX.matcher(xsdDateTime);
82  		if (mat.find()) {
83  			String tzCode = "GMT" + mat.group(1) + mat.group(2); // eg "GMT+0100"
84  			TimeZone tz = TimeZone.getTimeZone(tzCode);
85  			sdf.setTimeZone(tz);
86  		}
87  
88  		try {
89  			return sdf.parse(xsdDateTime);
90  		} catch (ParseException px) {
91  			throw new IllegalArgumentException(xsdDateTime + " is not xsd:dateTime");
92  		}
93  	}

226 	public static int getDaysDiff(Date older, Date newer, boolean absValue) {
227 		Calendar cal = Calendar.getInstance();
228 		cal.setTime(older);
229 		cal.set(Calendar.HOUR_OF_DAY, 0);
230 		cal.set(Calendar.MINUTE, 0);
231 		cal.set(Calendar.SECOND, 0);
232 		cal.set(Calendar.MILLISECOND, 0);
233 		long lOlder = cal.getTimeInMillis() + cal.getTimeZone().getOffset(cal.getTimeInMillis());
234 
235 		cal.setTime(newer);
236 		cal.set(Calendar.HOUR_OF_DAY, 0);
237 		cal.set(Calendar.MINUTE, 0);
238 		cal.set(Calendar.SECOND, 0);
239 		cal.set(Calendar.MILLISECOND, 0);
240 		long lNewer = cal.getTimeInMillis() + cal.getTimeZone().getOffset(cal.getTimeInMillis());
241 
242 		int diff = Math.round((lNewer - lOlder) / DAY_MILLIS);
243 		return absValue ? Math.abs(diff) : diff;
244 	}

251 	public static int getMonthsDiff(Date older, Date newer) {
252 		Calendar cal = Calendar.getInstance();
253 		cal.setTime(older);
254 		int rok1 = cal.get(Calendar.YEAR);
255 		int mes1 = cal.get(Calendar.MONTH);
256 		int den1 = cal.get(Calendar.DAY_OF_MONTH);
257 
258 		cal.setTime(newer);
259 		int rok2 = cal.get(Calendar.YEAR);
260 		int mes2 = cal.get(Calendar.MONTH);
261 		int den2 = cal.get(Calendar.DAY_OF_MONTH);
262 
263 		int diff = Math.abs(rok2 - rok1) * 12 + (mes2 - mes1);
264 		if (den1 <= den2) {
265 			return diff;
266 		} else { //den1 > den2 = necely mesic
267 			return diff - 1;
268 		}
269 	}
